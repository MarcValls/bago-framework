==== EJECUTAR EN EL LENOVO (PowerShell como administrador) ====

# 1. Verificar estado red
ipconfig /all

# 2. Si Ethernet deshabilitado:
netsh interface set interface "Ethernet" admin=enabled
netsh interface set interface "Conexión de área local" admin=enabled

# 3. Asignar IP APIPA fija (para encontrarnos):
netsh interface ip set address "Ethernet" static 169.254.31.1 255.255.0.0

# 4. Ping a este PC:
ping 169.254.31.155

# 5. Habilitar WinRM (para control remoto desde BAGO):
Enable-PSRemoting -Force -SkipNetworkProfileCheck
Set-Item WSMan:\localhost\Client\TrustedHosts -Value "*" -Force
netsh advfirewall firewall set rule group="Windows Remote Management" new enable=Yes

# 6. Verificar que funciona:
Test-NetConnection -ComputerName 169.254.31.155 -Port 5985
